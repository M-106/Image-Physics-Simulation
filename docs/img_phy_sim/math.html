img_phy_sim.math

Mathematical and Geometric Utility Functions

This module provides lightweight mathematical helper functions for 2D geometry and angle-based computations. It is designed to support higher-level operations such as beam tracing, image-based simulations, and coordinate normalization.

The functions focus on conversions between angular and Cartesian representations, as well as normalization and denormalization of image coordinates.

Main features:

  • Generate evenly spaced degree ranges for directional sampling
  • Convert between degrees and 2D unit vectors
  • Convert 2D vectors back to degree angles
  • Normalize and denormalize 2D points for image-based coordinate systems

Typical use cases:

  • Generating input directions for beam tracing or ray simulation
  • Converting between angular and vector representations in geometric algorithms
  • Preparing coordinates for normalized image processing workflows

Dependencies:

  • math
  • numpy

Example:

import img_phy_sim as ips

# Generate sample directions
directions = ips.math.get_linear_degree_range(step_size=45)

# Convert each to a 2D vector
vectors = [ips.math.degree_to_vector(d) for d in directions]

# Convert back to degrees
recovered = [ips.math.vector_to_degree(v) for v in vectors]

# Normalize and denormalize a point
p_norm = ips.math.normalize_point((128, 64), width=256, height=128)
p_pixel = ips.math.denormalize_point(p_norm, width=256, height=128)

 79def get_linear_degree_range(start=0, stop=360, step_size=10, offset=0):
 80    """
 81    Generate a list of degrees within a linear range.
 82
 83    Parameters:
 84    - start (int, optional): Starting degree (default is 0).
 85    - stop (int, optional): Ending degree (default is 360).
 86    - step_size (int, optional): Step size between degrees (default is 10).
 87    - offset (int, optional): Offset to add to each degree value (default is 0).
 88
 89    Returns:
 90    - list: List of degree values adjusted by offset and modulo 360.
 91    """
 92    degree_range = np.arange(start=start, stop=stop, step=step_size).tolist() # list(range(0, 360, step_size))
 93    return list(map(lambda x: (x+offset) % 360, degree_range))

 97def degree_to_vector(degree):
 98    """
 99    Convert a degree angle to a 2D unit vector.
100
101    Parameters:
102    - degree (float): Angle in degrees.
103
104    Returns:
105    - list: 2D vector [cos(degree), sin(degree)].
106    """
107    # rad = math.radians(degree)
108    # return [math.cos(rad), math.sin(rad)]
109    rad = degree * _DEG2RAD
110    return (math.cos(rad), math.sin(rad))

115def vector_to_degree(x, y):
116    """
117    Convert a 2D vector into its corresponding degree angle.
118
119    Parameters:
120    - vector (tuple): 2D vector (x, y).
121
122    Returns:
123    - int: Angle in degrees within the range [0, 360).
124    """
125    # x, y = vector
126    # degree = math.degrees(math.atan2(y, x))  # atan2 returns angles between -180° and 180°
127    # return int( degree % 360 ) 
128    deg = math.atan2(y, x) * _RAD2DEG
129    return int(deg % 360.0)

133def normalize_point(x, y, width, height):
134    """
135    Normalize a 2D point to the range [0, 1].
136
137    Parameters:
138    - point (tuple): (x, y) coordinates of the point.
139    - width (int): Image or grid width.
140    - height (int): Image or grid height.
141
142    Returns:
143    - tuple: Normalized point (x / (width - 1), y / (height - 1)).
144    """
145    # return (x / (width - 1), y / (height - 1))
146    return ( (1.0 / (width - 1.0))*x, 
147             (1.0 / (height - 1.0))*y)

152def denormalize_point(x, y, width, height):
153    """
154    Denormalize a 2D point from normalized coordinates back to pixel coordinates.
155
156    Parameters:
157    - point (tuple): Normalized (x, y) coordinates.
158    - width (int): Image or grid width.
159    - height (int): Image or grid height.
160
161    Returns:
162    - tuple: Denormalized point (x * (width - 1), y * (height - 1)).
163    """
164    return (x * (width - 1.0), y * (height - 1.0))

168def numpy_info(numpy_array, should_print=True):
169    """
170    Print and return basic statistical information about a NumPy array.
171
172    Computes common descriptive statistics such as mean, median, standard
173    deviation, variance, minimum, maximum, and array shape. This function is
174    primarily intended for quick debugging and inspection of intermediate
175    results in numerical or image-processing pipelines.
176
177    Parameters:
178    - numpy_array (np.ndarray): Input NumPy array to analyze.
179    - should_print (bool): If True, print the statistics to stdout.
180
181    Returns:
182    - str: A formatted string containing the computed statistics.
183    """
184    result = "Array Statistics:"
185    result += f"\n    - mean = {numpy_array.mean():.2f}"
186    result += f"\n    - median = {np.median(numpy_array):.2f}"
187    # result += f"\n    - percentile = {np.percentile(numpy_array, [25, 50, 75])}"
188    result += f"\n    - std = {numpy_array.std():.2f}"
189    result += f"\n    - var = {numpy_array.var():.2f}"
190    result += f"\n    - min = {numpy_array.min():.2f}"
191    result += f"\n    - max = {numpy_array.max():.2f}"
192    result += f"\n    - shape = {numpy_array.shape}"
193    
194    if should_print:
195        print(result)
196    return result

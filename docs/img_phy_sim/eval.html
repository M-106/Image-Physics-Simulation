Evaluation Utility Functions

  • Calculate Metrice between rays and a target ray-image

  • Evaluation / accuracy measurement

  • numpy
  • ips -> draw rays

import img_phy_sim as ips

dataset = ips.data.PhysGenDataset(mode='test', variation="sound_reflection", input_type="osm", output_type="complex_only")

f1_mean = 0
recall_mean = 0
precision_mean = 0
counter = 0
for (input_img, target_img, idx) in dataset:
    rays = ips.ray_tracing.trace_beams(rel_position=[0.5, 0.5], 
                                        img_src=input_img.squeeze(0).numpy(), 
                                        directions_in_degree=ips.math.get_linear_degree_range(start=0, stop=360, step_size=36),
                                        wall_values=None, 
                                        wall_thickness=0,
                                        img_border_also_collide=False,
                                        reflexion_order=3,
                                        should_scale_rays=True,
                                        should_scale_img=False)
    f1, recall, precision = ips.math.calc_metrices(rays, nm_gt, eval_name=f"{len(ips.math.get_linear_degree_range(start=0, stop=360, step_size=i))} Rays", should_print=False)
    f1_mean += f1
    recall_mean += recall
    precision_mean += precision
    counter += 1

f1_mean /= counter
recall_mean /= counter
precision_mean /= counter

print(f"Baseline Accuracy: F1={f1_mean:.2f}, Recall={recall_mean:.02f}, Precision={precision_mean:.02f}")

 73def calc_metrices(rays, noise_modelling_gt, rays_format_is_image=False, eval_name="", should_print=True):
 74    """
 75    Compute Precision, Recall, and F1 score by comparing ray-based predictions
 76    against a ground truth image.
 77
 78    The function converts a set of rays into a binary image representation
 79    (if not already provided as an image), normalizes both prediction and
 80    ground truth if necessary, and evaluates their pixel-wise overlap.
 81    All non-zero pixels are treated as positives.
 82
 83    Parameters:
 84    - rays (array-like or np.ndarray):<br>
 85        Ray representation or pre-rendered ray image, depending on
 86        `rays_format_is_image`.
 87    - noise_modelling_gt (np.ndarray):<br>
 88        Ground truth image used for evaluation.
 89    - rays_format_is_image (bool):<br>
 90        If True, `rays` is assumed to already be an image.
 91        If False, rays are rendered into an image using `draw_rays`.
 92    - eval_name (str):<br>
 93        Optional identifier printed alongside the evaluation results.
 94    - should_print (bool):<br>
 95        If True, print the computed metrics to stdout.
 96
 97    Returns:
 98    - tuple:<br>
 99        (f1, recall, precision) computed from binary pixel overlap.
100    """
101    # Create image from rays
102    if rays_format_is_image:
103        ray_img = rays
104    else:
105        ray_img = draw_rays(rays, detail_draw=True, 
106                            output_format="single_image", 
107                            img_background=None, ray_value=1.0, ray_thickness=1, 
108                            img_shape=(256, 256), dtype=float, standard_value=0,
109                            should_scale_rays_to_image=True, original_max_width=None, original_max_height=None)
110    
111    # Normalize both (if needed)
112    if (noise_modelling_gt > 1.0).any():
113        # raise ValueError("Noise Modelling Ground Truth Image is not normalized.")
114        noise_modelling_gt /= 255
115
116    if (ray_img > 1.0).any():
117        # raise ValueError("Ray Image is not normalized.")
118        ray_img /= 255
119
120    # Thresholding to binary images
121    noise_modelling_gt_binary = noise_modelling_gt != 0.0
122    # numpy_info(noise_modelling_gt_binary)
123    rays_binary = ray_img != 0.0
124
125    # Recall, Precision, F1 Score
126    overlap = noise_modelling_gt_binary * rays_binary
127
128    #     recall - how is the coverage towards the gt?
129    recall = np.sum(overlap) / np.sum(noise_modelling_gt_binary)
130
131    #     precision - how many rays hit the right place?
132    precision = np.sum(overlap) / np.sum(rays_binary)
133
134    #     f1
135    f1 = 2*(precision*recall) / (precision+recall)
136
137    if should_print:
138        print(f"Eval {eval_name}: F1={f1:.02f}, Recall={recall:.02f}, Precision={precision:.02f}")
139    return f1, recall, precision
